      import java.text.DecimalFormat;
      import java.text.DecimalFormatSymbols;
      import java.text.NumberFormat;
      import java.util.Locale;
      import java.util.Scanner;
      public class Main {
      
          /**
           * @param args the command line arguments
           */
          public static void main(String[] args) {
      
              
              int numeroFuncionario = 0;
              int qtdHorasTrabalhadas = 0;
              double valorHora = 0;
              double salario = 0;
              String padrao = "##,####.00";
              
              Scanner scan = new Scanner(System.in);
              DecimalFormat df = new DecimalFormat(padrao);
              
              //instanciando um objeto da Classe DecimaFormatSymbols para alterar o separador de casa decimal
              DecimalFormatSymbols formatandoSimbolo = new DecimalFormatSymbols(new Locale("pt", "Brazil"));
              formatandoSimbolo.setDecimalSeparator('.');
              formatandoSimbolo.setGroupingSeparator(',');
      
              //System.out.println("Digite o numero de identificação : ");
              numeroFuncionario = scan.nextInt();
              
              //System.out.println("Digite a quatidade de horas trabalhadas : ");
              qtdHorasTrabalhadas = scan.nextInt();
              
              //System.out.println("Digite a nota C do aluno : ");
              valorHora = scan.nextDouble();
              
              //System.out.println("Digite a nota D do aluno : ");
              salario = qtdHorasTrabalhadas * valorHora;
              
              System.out.println("NUMBER = "+ numeroFuncionario + "\nSALARY = U$ " + df.format(salario));
      
          }//fim do method main
          
      }//fim da class
