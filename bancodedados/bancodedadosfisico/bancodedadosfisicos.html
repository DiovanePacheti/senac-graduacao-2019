                        CREATE DATABASE IF NOT EXISTS empregado;
                        USE empregado;
                        
                        CREATE TABLE empregado (
                            codigo Integer NOT NULL,
                            nome CHAR(40) NOT NULL,
                            setor CHAR(2),
                            cargo CHAR(20),
                            salario DECIMAL(10,2),
                            PRIMARY KEY (codigo)

Apresentando setores e nomes colocando por ordem ascendente do campo setor e descendente do campo nome
