Código em JavaScript

function calculandoProdutoMercado(){

    var inProduto = document.getElementById("inProduto");
    var inPreco = document.getElementById("inPreco");


    var produto = inProduto.value;
    var preco = Number(inPreco.value);

    var valorTotalComDescontoNaTerceiraUnidade = 
    (preco * 2) + (preco - (preco * 0.50) );

    var outSaidaResultadoProduto = document.getElementById
    ("outSaidaResultadoProduto");
    outSaidaResultadoProduto.textContent = "Promoção " + produto;

    var outSaidaResultadoPreco = document.getElementById
    ("outSaidaResultadoPreco");
    outSaidaResultadoPreco.textContent = "Leve 3 por apens " 
    + valorTotalComDescontoNaTerceiraUnidade.toFixed(2) + " R$";


}//fim da function calculando produto mercado

var btCalcular = document.getElementById("btCalcular");
btCalcular.addEventListener("click", calculandoProdutoMercado);

package mercadoavenida.com;
import java.util.Scanner;
public class MercadoAvenida {

  public static void main(String[] args) {

    String produto = " ";
    double preco = 0;
    double descontoTerceiraUnidade = 0;
        
    Scanner scan = new Scanner(System.in);
    
    System.out.print("Digite o Produto : ");
    produto = scan.next();
    
    System.out.print("Digite o preço : ");
    preco = scan.nextDouble();
    
    descontoTerceiraUnidade = (preco * 2) +
    (preco - (preco * 0.50));
    
    System.out.println("Promoção : " + produto);
    System.out.println("Leve 3 por apenas " +
    descontoTerceiraUnidade + " R$");

  }//fim do method

}//fim da class
